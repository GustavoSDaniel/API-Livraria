

CREATE TABLE AUTOR (
    ID BIGINT NOT NULL PRIMARY KEY,
    NOME VARCHAR(100) NOT NULL,
    DATA_NASCIMENTO DATE NOT NULL,
    NACIONALIDADE VARCHAR(50) NOT NULL
);


CREATE TABLE LIVRO (
    ID BIGINT NOT NULL PRIMARY KEY,
    ISBN VARCHAR(20) NOT NULL,
    TITULO VARCHAR(150) NOT NULL,
    DATA_PUBLICACAO DATE NOT NULL, -- Corrigido o tipo de dado
    GENERO ENUM('FICCAO', 'FANTASIA', 'MISTERIO', 'ROMANCE', 'BIOGRAFIA', 'CIENCIA') NOT NULL, -- Substitui CHECK por ENUM
    PRECO DECIMAL(18,2), -- DECIMAL Ã© mais comum que NUMERIC no MySQL
    ID_AUTOR BIGINT NOT NULL,
    CONSTRAINT FK_AUTOR FOREIGN KEY (ID_AUTOR) REFERENCES AUTOR(ID) -- Forma correta de definir chave estrangeira
);
